泛型的概念

允许在定义类、接口时通过一个标识表示类中某个属性的类型或者是某个方法的返回值及参数类型。

  • 类型的参数化，就是可以把类型像方法的参数那样传递。
  • 泛型使编译器可以在编译期间对类型进行检查以提高类型安全，减少运行时由于对象类型不匹配引发的异常。
    @Test
    public void test1(){
        ArrayList<Integer> list = new ArrayList<Integer>();
        list.add(111);
        list.add(222);
        list.add(333);
//        for (Integer num:list){
//            System.out.println(num);
//        }
        Iterator<Integer> iterator = list.iterator();
        while(iterator.hasNext()){
            System.out.println(iterator.next());
        }
    }

    @Test
    public void test2(){
        HashMap<String,Integer> map = new HashMap<String,Integer>();
        map.put("Tony",18);
        map.put("Troye",25);
        map.put("Taylor",20);
        Set<Map.Entry<String, Integer>> entries = map.entrySet();
        Iterator<Map.Entry<String, Integer>> iterator = entries.iterator();
        while(iterator.hasNext()){
            Map.Entry<String,Integer> entry = iterator.next();
            System.out.println(entry .getKey()+"-----"+entry .getValue());
        }
    }

List<?>、List<? extends T> 与 List<? super T>

public class Test3 {
    public static void main(String[] args) {
        List list = new ArrayList();
        list.add(new Object());
        list.add(new Integer(111));
        list.add(new String("aaa"));
        Object o = list.get(0);

        // List<?> 允许添加元素，不允许添加元素
        List<?> list1 = list;
        list1.remove(0);
        list1.remove(new Integer(111));
        // list1.add(new String("bbb"));
        Object o1 = list1.get(0);
        list1.stream().forEach(System.out::println);

        ArrayList<Person> persons = new ArrayList<>();
        persons.add(new Person());
        persons.add(new Male());
        persons.add(new Female());
        // List<? extends Person> 允许get元素，不允许添加元素
        List<? extends Person> persons1 = persons;
        // persons1.add(new Person());
        // persons1.add(new Male());
        // persons1.add(new Female());
        Person person = persons1.get(0);

        // List<? super Person> 允许添加元素，不允许get元素
        List<? super Person> persons2 = new ArrayList<Person>();
        persons2.add(new Male());
        persons2.add(new Female());
        persons2.add(new Person());
        // Person person = persons2.get(0);

    }
}
