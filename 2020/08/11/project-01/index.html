创建emp数据库，创建表

create table t_user(
	id int(6) primary key auto_increment,
	username varchar(50),
	realname varchar(50),
    password varchar(50),
    gender varchar(4),
    status varchar(4),
    regist_time timestamp
);

create table t_emp(
	id int(6) primary key auto_increment,
    name varchar(50),
    photo_path varchar(100),
    salary double(10,2),
    age int(3)
)

2、编写springboot配置文件
