冒泡排序

public class BubbleSort {
    public static void main(String[] args) {
        int[] arr = {1, 2, 3, 4, 6, 5};
        /**
         *  算法思路：
         *  假设从小到大排列：
         *  对数组中的每个数进行两两比较，将如果前一个数大于后一个数，则交换位置，
         *  这样一趟比较下来，最大的那个数就排到了最后
         *  也就是说，i个数据进行 i-1 趟比较就完成了排序
         */

        // 第1趟  5次比较
       /* for (int i = 0; i < 5; i++) {
            if (arr[i] > arr[i + 1]) {
                int temp = arr[i];
                arr[i] = arr[i + 1];
                arr[i + 1] = temp;
            }
        }
        System.out.println(Arrays.toString(arr));// 最大的数排到了最后[5, 4, 3, 2, 1, 6]*/
        // 第2趟  4次比较
        // 第3趟  3次比较
        // ......
        // 第n-1趟  6-(i-1)=5-n次比较  =>arr.length-1-i次

        // 将两层循环嵌套
        /**
         * 改进：
         * 假如一趟比较下来，没有发生过一次交换，那么可以认为数组已经排好序列了
         * 可以设置一个标志位，发生了比较就修改标志位，如果在某趟比较中没有发生交换，则直接结束循环
         */
        boolean flag;
        for (int i = 0; i < arr.length - 1; i++) {
            flag = true;
            for (int j = 0; j < arr.length - 1 - i; j++) {
                if (arr[j] > arr[j + 1]) {
                    flag = false;
                    int temp = arr[j];
                    arr[j] = arr[j + 1];
                    arr[j + 1] = temp;
                }
            }
            if(flag){
                break;
            }
            System.out.println("第" + (i + 1) + "趟比较：" + Arrays.toString(arr));
        }
        System.out.println("最终结果：" + Arrays.toString(arr));
    }
}

快速排序

public class QuickSort {
    public static void main(String[] args) {
        int[] arr = {6,5,4,4,3,2,1};
        quickSort(arr,0,6);
        System.out.println(Arrays.toString(arr));
        Arrays.sort(arr);
    }
    /**
     * 通过一趟排序将要排序的数据分割成独立的两部分，其中一部分的所有数据都比另外一部分的所有数据都要小
     * @param arr
     * @param l 数组第一个元素的下标
     * @param r 数组最后一个元素的下标
     */
    public static void quickSort(int[] arr, int l, int r) {
        int i = l;// 左下标
        int j = r;// 右下标
        int pivot = arr[(l+r)/2];// 中间的元素作为基准
        while(i<=j){
            // 在支点左边寻找比基准大的数
            while(arr[i]<pivot){
                i++;
            }
            // 在支点右边寻找比基准小的数
            while(arr[j]>pivot){
                j--;
            }
            // 此时已经分别找到了比基准小的数(右边)、比基准大的数(左边)，将它们进行交换
            if(i<=j){
                int temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
                i++;
                j--;
            }
        }
        // 上面一个while保证了基准的左边比支点小，基准的右边比支点大
        // 左边再做排序，直到左边剩下一个数(递归出口)
        if (l < j){
            quickSort(arr, l, j);
        }
        // 右边再做排序，直到右边剩下一个数(递归出口)
        if (i < r){
            quickSort(arr, i, r);
        }
    }
}

希尔排序

public class ShellSort {
    public static void main(String[] args) {
        int[] arr = {5,4,3,2,1};
        /**
         * 将序列进行分组，设置一个步长，这个步长就等于分了多少组，然后将分的这些组进行简单插入排序
         * 这样每个组之间就是有序的了
         * 进一步对序列分组，直到分成一个组
         * 则整个序列就有序了
         */
        for (int step = arr.length/2; step>0 ; step/=2) {// 分组
            for(int i = step;i<arr.length;i++){ // 对分组进行简单插入排序
                int temp = arr[i];// 记录当前元素
                int j = i-step;// 记录与比较的元素的下标
                while(j>=0&&temp<arr[j]){
                    arr[j+step] = arr[j];
                    j-=step;
                }
                // 找到插入位置
                arr[j+step] = temp;
            }
        }
        System.out.println(Arrays.toString(arr));
    }
}

二分查找

public class BinarySearch {
    public static void main(String[] args) {
        int[] arr = {1,2,3,4,5,6};// 二分查找序列必须有序
        int index = binarySearch(arr,0,arr.length,1);
        System.out.println(index);
    }

    /**
     *
     * @param arr
     * @param left 左索引
     * @param right 右索引
     * @param val 要查找的值
     * @return 查找成功返回下标，否则返回-1
     */
    public static int binarySearch(int[] arr,int left,int right,int val){
        if(left>right){
            return -1;
        }
        int mid = (left+right)/2;
        int midVal = arr[mid];
        if(val<midVal){// 向左递归
            return binarySearch(arr,left,mid-1,val);
        }else if(val>midVal){// 向右递归
            return binarySearch(arr,mid+1,right,val);
        }else {
            return mid;
        }
    }
}
